```go
	for i, docID := range m.docs {
		v.VisitValue(docID, m.values[i])
	}

// collectVisitor implements IntersectVisitor and collects matching docIDs.
type collectVisitor struct {
	min, max int64
	docs     []int
}

func (v *collectVisitor) Visit(docID int) {
	v.docs = append(v.docs, docID)
}

func (v *collectVisitor) VisitValue(docID int, value int64) {
	if value >= v.min && value <= v.max {
		v.docs = append(v.docs, docID)
	}
}

func (v *collectVisitor) Compare(minValue, maxValue int64) Relation {
	if maxValue < v.min || minValue > v.max {
		return CellOutsideQuery
	}
	if minValue >= v.min && maxValue <= v.max {
		return CellInsideQuery
	}
	return CellCrossesQuery
}
```

```go
package bkd

// MaxPointsInLeafNode is the maximum number of points stored in a single leaf.
// Matches Lucene's default BKDConfig.DEFAULT_MAX_POINTS_IN_LEAF_NODE.
const MaxPointsInLeafNode = 512

// Relation describes how a BKD tree cell relates to a query range.
type Relation int

const (
	CellOutsideQuery Relation = iota // No overlap — skip subtree
	CellInsideQuery                  // Fully contained — bulk visit all docs
	CellCrossesQuery                 // Partial overlap — recurse or filter
)

// IntersectVisitor guides BKD tree traversal via the visitor pattern.
// Matches Lucene's PointValues.IntersectVisitor.
type IntersectVisitor interface {
	// Visit is called for docs in cells fully inside the query.
	Visit(docID int)
	// VisitValue is called for docs in cells crossing the query boundary.
	VisitValue(docID int, value int64)
	// Compare determines the relation between a cell's value bounds and the query.
	Compare(minValue, maxValue int64) Relation
}

// PointTree provides navigation over a BKD tree hierarchy.
// Matches Lucene's PointValues.PointTree.
type PointTree interface {
	MoveToChild() bool
	MoveToSibling() bool
	MoveToParent() bool
	MinValue() int64
	MaxValue() int64
	Size() int
	VisitDocIDs(v IntersectVisitor)
	VisitDocValues(v IntersectVisitor)
}

// Intersect traverses a PointTree using the visitor to guide pruning.
// Matches Lucene's PointValues.intersect().
func Intersect(tree PointTree, visitor IntersectVisitor) {
	r := visitor.Compare(tree.MinValue(), tree.MaxValue())
	switch r {
	case CellOutsideQuery:
		return
	case CellInsideQuery:
		tree.VisitDocIDs(visitor)
	case CellCrossesQuery:
		if tree.MoveToChild() {
			Intersect(tree, visitor)
			if tree.MoveToSibling() {
				Intersect(tree, visitor)
			}
			tree.MoveToParent()
		} else {
			tree.VisitDocValues(visitor)
		}
	}
}
```

```go
import (
	"encoding/binary"
	"fmt"
	"math"
	"sort"

	"gosearch/store"
)

// point holds a docID-value pair for tree construction.
type point struct {
	docID int
	value int64
}

// BKDWriter builds a BKD tree from point values and writes it to a .kd file.
type BKDWriter struct {
	points []point
}

// NewBKDWriter creates a new BKDWriter.
func NewBKDWriter() *BKDWriter {
	return &BKDWriter{}
}

// Add buffers a point for later tree construction.
func (w *BKDWriter) Add(docID int, value int64) {
	w.points = append(w.points, point{docID: docID, value: value})
}

// Finish builds the BKD tree and writes it to a .kd file.
// File: {segName}.{field}.kd
func (w *BKDWriter) Finish(dir store.Directory, segName, field string) error {
	fileName := fmt.Sprintf("%s.%s.kd", segName, field)
	out, err := dir.CreateOutput(fileName)
	if err != nil {
		return fmt.Errorf("create %s: %w", fileName, err)
	}
	defer out.Close()

	numPoints := len(w.points)
	if numPoints == 0 {
		return w.writeEmpty(out)
	}

	// Sort all points by value
	sort.Slice(w.points, func(i, j int) bool {
		return w.points[i].value < w.points[j].value
	})

	// Count distinct docs
	docSet := make(map[int]struct{}, numPoints)
	for _, p := range w.points {
		docSet[p.docID] = struct{}{}
	}
	docCount := len(docSet)

	// Calculate tree structure
	numLeaves := (numPoints + MaxPointsInLeafNode - 1) / MaxPointsInLeafNode
	// Round up to next power of 2 for balanced binary heap
	numLeaves = nextPowerOf2(numLeaves)
	numInnerNodes := numLeaves - 1

	// Build leaf assignments by recursive splitting
	leaves := make([][]point, numLeaves)
	w.buildLeaves(w.points, leaves, 1, numInnerNodes)

	// Build inner node split values and subtree sizes
	innerNodes := make([]innerNodeEntry, numInnerNodes+1) // 1-indexed
	w.buildInnerNodes(leaves, innerNodes, 1, 0, numLeaves-1)

	// Compute leaf metadata
	type leafMeta struct {
		offset   uint64
		numPts   int
		minValue int64
		maxValue int64
	}
	leafMetas := make([]leafMeta, numLeaves)
	var dataOffset uint64
	for i, leaf := range leaves {
		n := len(leaf)
		leafMetas[i].numPts = n
		leafMetas[i].offset = dataOffset
		if n > 0 {
			leafMetas[i].minValue = leaf[0].value
			leafMetas[i].maxValue = leaf[n-1].value
		}
		// Each leaf: n×uint32 (docIDs) + n×int64 (values)
		dataOffset += uint64(n*4 + n*8)
	}

	globalMin := w.points[0].value
	globalMax := w.points[numPoints-1].value

	// Write header (32 bytes)
	if err := out.WriteUint32(uint32(MaxPointsInLeafNode)); err != nil {
		return err
	}
	if err := out.WriteUint32(uint32(numLeaves)); err != nil {
		return err
	}
	if err := out.WriteUint32(uint32(numPoints)); err != nil {
		return err
	}
	if err := out.WriteUint32(uint32(docCount)); err != nil {
		return err
	}
	if err := out.WriteUint64(uint64(globalMin)); err != nil {
		return err
	}
	if err := out.WriteUint64(uint64(globalMax)); err != nil {
		return err
	}

	// Write inner nodes (1-indexed, skip index 0)
	for i := 1; i <= numInnerNodes; i++ {
		if err := out.WriteUint64(uint64(innerNodes[i].splitValue)); err != nil {
			return err
		}
		if err := out.WriteUint32(uint32(innerNodes[i].numPoints)); err != nil {
			return err
		}
	}

	// Write leaf directory
	for _, lm := range leafMetas {
		if err := out.WriteUint64(lm.offset); err != nil {
			return err
		}
		if err := out.WriteUint32(uint32(lm.numPts)); err != nil {
			return err
		}
		if err := out.WriteUint64(uint64(lm.minValue)); err != nil {
			return err
		}
		if err := out.WriteUint64(uint64(lm.maxValue)); err != nil {
			return err
		}
	}

	// Write leaf data
	for _, leaf := range leaves {
		for _, p := range leaf {
			if err := out.WriteUint32(uint32(p.docID)); err != nil {
				return err
			}
		}
		buf := make([]byte, 8)
		for _, p := range leaf {
			binary.LittleEndian.PutUint64(buf, uint64(p.value))
			if _, err := out.Write(buf); err != nil {
				return err
			}
		}
	}

	return nil
}

		return err
	}
	if err := out.WriteUint32(0); err != nil { // numLeaves
		return err
	}
	if err := out.WriteUint32(0); err != nil { // numPoints
		return err

	if err := out.WriteUint32(0); err != nil { // docCount
		return err

	if err := out.WriteUint64(0); err != nil { // globalMin
		return err
	}
	if err := out.WriteUint64(0); err != nil { // globalMax
		return err
	}
	return nil

	splitValue int64
	numPoints  int
}

// buildLeaves recursively splits sorted points into leaf buckets.
// nodeID is the 1-based heap index; when nodeID > numInnerNodes, it's a leaf.
func (w *BKDWriter) buildLeaves(points []point, leaves [][]point, nodeID, numInnerNodes int) {
	if nodeID > numInnerNodes {
		// Leaf: nodeID - numInnerNodes - 1 is the leaf index
		leafIdx := nodeID - numInnerNodes - 1
		leaves[leafIdx] = points
		return
	}
	// Split at midpoint
	mid := len(points) / 2
	w.buildLeaves(points[:mid], leaves, nodeID*2, numInnerNodes)
	w.buildLeaves(points[mid:], leaves, nodeID*2+1, numInnerNodes)
}

func (w *BKDWriter) buildInnerNodes(leaves [][]innerNodeEntry, nodes []innerNodeEntry, nodeID, leafStart, leafEnd int) {
	// This is handled differently — we compute from the leaf assignments
}

// Actually, let's compute inner nodes after leaf assignment:
func (w *BKDWriter) computeInnerNodes(leaves [][]point, nodes []innerNodeEntry, nodeID, leafStart, leafEnd int) {
	if leafStart == leafEnd {
		// Leaf node — not an inner node
		return
	}
	mid := (leafStart + leafEnd) / 2

	// Split value: max value in the left subtree
	leftMax := int64(math.MinInt64)
	for i := leafStart; i <= mid; i++ {
		if len(leaves[i]) > 0 && leaves[i][len(leaves[i])-1].value > leftMax {
			leftMax = leaves[i][len(leaves[i])-1].value
		}
	}
	nodes[nodeID].splitValue = leftMax

	// Total points in subtree
	total := 0
	for i := leafStart; i <= leafEnd; i++ {
		total += len(leaves[i])
	}
	nodes[nodeID].numPoints = total

	// Recurse
	w.computeInnerNodes(leaves, nodes, nodeID*2, leafStart, mid)
	w.computeInnerNodes(leaves, nodes, nodeID*2+1, mid+1, leafEnd)
}

// nextPowerOf2 returns the smallest power of 2 >= n. Returns 1 for n <= 1.
func nextPowerOf2(n int) int {
	if n <= 1 {
		return 1
	}
	p := 1
	for p < n {
		p *= 2
	}
	return p
}
```

```go
	// Split value = max value in left subtree's leaves
	leftMax := int64(math.MinInt64)
	for i := leafStart; i <= mid; i++ {
		if n := len(leaves[i]); n > 0 {
			if leaves[i][n-1].value > leftMax {
				leftMax = leaves[i][n-1].value
			}
		}
	}
	nodes[nodeID].splitValue = leftMax

	total := 0
	for i := leafStart; i <= leafEnd; i++ {
		total += len(leaves[i])
	}
	nodes[nodeID].numPoints = total

	w.computeInnerNodes(leaves, nodes, nodeID*2, leafStart, mid)
	w.computeInnerNodes(leaves, nodes, nodeID*2+1, mid+1, leafEnd)
}

func nextPowerOf2(n int) int {
	if n <= 1 {
		return 1
	}
	p := 1
	for p < n {
		p *= 2
	}
	return p
}
```

```go
const (
	headerSize        = 32 // 4+4+4+4+8+8
	innerNodeSize     = 12 // 8 (splitValue) + 4 (numPoints)
	leafDirEntrySize  = 28 // 8 (offset) + 4 (numPoints) + 8 (min) + 8 (max)
)

type leafDirEntry struct {
	offset   uint64
	numPts   int
	minValue int64
	maxValue int64
}

// BKDReader reads a .kd file and provides PointTree access for range queries.
type BKDReader struct {
	data          *store.MMapIndexInput
	numLeaves     int
	numPoints     int
	docCount      int
	numInnerNodes int
	globalMin     int64
	globalMax     int64
	innerNodes    []innerNodeEntry // 1-indexed, size = numInnerNodes+1
	leafDir       []leafDirEntry
	leafDataStart int // byte offset where leaf data begins
}

// OpenBKDReader opens a .kd file for the given segment and field.
func OpenBKDReader(dir store.Directory, segName, field string) (*BKDReader, error) {
	fileName := fmt.Sprintf("%s.%s.kd", segName, field)
	path := dir.FilePath(fileName)
	data, err := store.OpenMMap(path)
	if err != nil {
		return nil, fmt.Errorf("open %s: %w", fileName, err)
	}

	r := &BKDReader{data: data}
	if err := r.readHeader(); err != nil {
		data.Close()
		return nil, err
	}
	return r, nil
}

func (r *BKDReader) readHeader() error {
	if r.data.Length() < headerSize {
		// Empty or minimal file
		r.numLeaves = 0
		r.numPoints = 0
		return nil
	}

	r.data.Seek(0)
	maxPtsLeaf, err := r.data.ReadUint32()
	if err != nil {
		return err
	}
	_ = maxPtsLeaf // validated by writer

	nl, err := r.data.ReadUint32()
	if err != nil {
		return err
	}
	r.numLeaves = int(nl)

	np, err := r.data.ReadUint32()
	if err != nil {
		return err
	}
	r.numPoints = int(np)

	dc, err := r.data.ReadUint32()
	if err != nil {
		return err
	}
	r.docCount = int(dc)

	gmin, err := r.data.ReadUint64()
	if err != nil {
		return err
	}
	r.globalMin = int64(gmin)

	gmax, err := r.data.ReadUint64()
	if err != nil {
		return err
	}
	r.globalMax = int64(gmax)

	if r.numLeaves == 0 {
		return nil
	}

	r.numInnerNodes = r.numLeaves - 1

	// Read inner nodes
	r.innerNodes = make([]innerNodeEntry, r.numInnerNodes+1)
	for i := 1; i <= r.numInnerNodes; i++ {
		sv, err := r.data.ReadUint64()
		if err != nil {
			return fmt.Errorf("read inner node %d: %w", i, err)
		}
		np, err := r.data.ReadUint32()
		if err != nil {
			return fmt.Errorf("read inner node %d size: %w", i, err)
		}
		r.innerNodes[i] = innerNodeEntry{splitValue: int64(sv), numPoints: int(np)}
	}

	// Read leaf directory
	r.leafDir = make([]leafDirEntry, r.numLeaves)
	for i := range r.numLeaves {
		off, err := r.data.ReadUint64()
		if err != nil {
			return fmt.Errorf("read leaf dir %d: %w", i, err)
		}
		np, err := r.data.ReadUint32()
		if err != nil {
			return fmt.Errorf("read leaf dir %d numPts: %w", i, err)
		}
		minV, err := r.data.ReadUint64()
		if err != nil {
			return fmt.Errorf("read leaf dir %d min: %w", i, err)
		}
		maxV, err := r.data.ReadUint64()
		if err != nil {
			return fmt.Errorf("read leaf dir %d max: %w", i, err)
		}
		r.leafDir[i] = leafDirEntry{
			offset:   off,
			numPts:   int(np),
			minValue: int64(minV),
			maxValue: int64(maxV),
		}
	}

	// Leaf data starts after header + inner nodes + leaf directory
	r.leafDataStart = headerSize + r.numInnerNodes*innerNodeSize + r.numLeaves*leafDirEntrySize

	return nil
}

// bkdPointTree is a stateful cursor navigating the BKD tree.
type bkdPointTree struct {
	reader    *BKDReader
	nodeID    int
	level     int
	nodeStack []int // parent nodeIDs for MoveToParent
}

func (t *bkdPointTree) isLeaf() bool {
	return t.nodeID > t.reader.numInnerNodes
}

func (t *bkdPointTree) leafIndex() int {
	return t.nodeID - t.reader.numInnerNodes - 1
}

func (t *bkdPointTree) MoveToChild() bool {
	if t.isLeaf() {
		return false
	}
	t.nodeStack = append(t.nodeStack, t.nodeID)
	t.nodeID = t.nodeID * 2
	t.level++
	return true
}

func (t *bkdPointTree) MoveToSibling() bool {
	if t.nodeID%2 == 1 {
		// Already right child — no sibling
		return false
	}
	// Move from left child to right child
	t.nodeID++
	return true
}

func (t *bkdPointTree) MoveToParent() bool {
	if len(t.nodeStack) == 0 {
		return false
	}
	t.nodeID = t.nodeStack[len(t.nodeStack)-1]
	t.nodeStack = t.nodeStack[:len(t.nodeStack)-1]
	t.level--
	return true
}

func (t *bkdPointTree) MinValue() int64 {
	if t.isLeaf() {
		ld := t.reader.leafDir[t.leafIndex()]
		return ld.minValue
	}
	// Walk down left children to leftmost leaf
	nodeID := t.nodeID
	for nodeID <= t.reader.numInnerNodes {
		nodeID = nodeID * 2
	}
	return t.reader.leafDir[nodeID-t.reader.numInnerNodes-1].minValue
}

func (t *bkdPointTree) MaxValue() int64 {
	if t.isLeaf() {
		ld := t.reader.leafDir[t.leafIndex()]
		return ld.maxValue
	}
	// Walk down right children to rightmost leaf
	nodeID := t.nodeID
	for nodeID <= t.reader.numInnerNodes {
		nodeID = nodeID*2 + 1
	}
	return t.reader.leafDir[nodeID-t.reader.numInnerNodes-1].maxValue
}

func (t *bkdPointTree) Size() int {
	if t.isLeaf() {
		return t.reader.leafDir[t.leafIndex()].numPts
	}
	return t.reader.innerNodes[t.nodeID].numPoints
}

func (t *bkdPointTree) VisitDocIDs(v IntersectVisitor) {
	if t.isLeaf() {
		t.visitLeafDocIDs(t.leafIndex(), v)
		return
	}
	// Visit all leaves under this subtree
	t.visitSubtreeDocIDs(t.nodeID, v)
}

func (t *bkdPointTree) visitSubtreeDocIDs(nodeID int, v IntersectVisitor) {
	if nodeID > t.reader.numInnerNodes {
		t.visitLeafDocIDs(nodeID-t.reader.numInnerNodes-1, v)
		return
	}
	t.visitSubtreeDocIDs(nodeID*2, v)
	t.visitSubtreeDocIDs(nodeID*2+1, v)
}

func (t *bkdPointTree) visitLeafDocIDs(leafIdx int, v IntersectVisitor) {
	ld := t.reader.leafDir[leafIdx]
	if ld.numPts == 0 {
		return
	}
	baseOffset := t.reader.leafDataStart + int(ld.offset)
	for i := range ld.numPts {
		docID, _ := t.reader.data.ReadUint32At(baseOffset + i*4)
		v.Visit(int(docID))
	}
}

func (t *bkdPointTree) VisitDocValues(v IntersectVisitor) {
	if !t.isLeaf() {
		return
	}
	leafIdx := t.leafIndex()
	ld := t.reader.leafDir[leafIdx]
	if ld.numPts == 0 {
		return
	}
	baseOffset := t.reader.leafDataStart + int(ld.offset)
	valuesOffset := baseOffset + ld.numPts*4 // values start after docIDs
	for i := range ld.numPts {
		docID, _ := t.reader.data.ReadUint32At(baseOffset + i*4)
		valBytes := t.reader.data.SliceBytes(valuesOffset+i*8, 8)
		value := int64(binary.LittleEndian.Uint64(valBytes))
		v.VisitValue(int(docID), value)
	}
}
```

```go
import (
	"gosearch/document"
	"gosearch/index"
	"gosearch/index/bkd"
	"sort"
)

// PointRangeQuery matches documents where a numeric point field's value falls
// within an inclusive [min, max] range.
type PointRangeQuery struct {
	field string
	Min   int64
	Max   int64
}

func NewPointRangeQuery(field string, min, max int64) *PointRangeQuery {
	return &PointRangeQuery{field: field, Min: min, Max: max}
}

// NewDoublePointRangeQuery creates a range query for double point fields.
func NewDoublePointRangeQuery(field string, min, max float64) *PointRangeQuery {
	return &PointRangeQuery{
		field: field,
		Min:   document.DoubleToSortableLong(min),
		Max:   document.DoubleToSortableLong(max),
	}
}

func (q *PointRangeQuery) ExtractTerms() []FieldTerm {
	return nil
}

func (q *PointRangeQuery) CreateWeight(searcher *IndexSearcher, scoreMode ScoreMode) Weight {
	return &pointRangeWeight{query: q}
}

type pointRangeWeight struct {
	query *PointRangeQuery
}

func (w *pointRangeWeight) Query() Query {
	return w.query
}

func (w *pointRangeWeight) Scorer(ctx index.LeafReaderContext) Scorer {
	seg := ctx.Segment

	if _, ok := seg.PointFields()[w.query.field]; !ok {
		return nil
	}

	pv := seg.PointValues(w.query.field)
	if pv == nil {
		return nil
	}

	// Intersect BKD tree to collect matching docIDs
	visitor := &pointRangeVisitor{
		min: w.query.Min,
		max: w.query.Max,
	}
	tree := pv.PointTree()
	bkd.Intersect(tree, visitor)

	if len(visitor.docs) == 0 {
		return nil
	}

	// Sort docIDs for ordered iteration
	sort.Ints(visitor.docs)

	return &pointRangeScorer{
		docs:     visitor.docs,
		liveDocs: seg.LiveDocs(),
		pos:      -1,
		doc:      -1,
	}
}

// pointRangeVisitor implements bkd.IntersectVisitor for range queries.
type pointRangeVisitor struct {
	min, max int64
	docs     []int
}

func (v *pointRangeVisitor) Visit(docID int) {
	v.docs = append(v.docs, docID)
}

func (v *pointRangeVisitor) VisitValue(docID int, value int64) {
	if value >= v.min && value <= v.max {
		v.docs = append(v.docs, docID)
	}
}

func (v *pointRangeVisitor) Compare(minValue, maxValue int64) bkd.Relation {
	if maxValue < v.min || minValue > v.max {
		return bkd.CellOutsideQuery
	}
	if minValue >= v.min && maxValue <= v.max {
		return bkd.CellInsideQuery
	}
	return bkd.CellCrossesQuery
}

// pointRangeScorer iterates over pre-collected matching docIDs.
type pointRangeScorer struct {
	docs     []int
	liveDocs *index.Bitset
	pos      int
	doc      int
}

func (s *pointRangeScorer) Score() float64 {
	return 1.0
}

func (s *pointRangeScorer) DocID() int {
	return s.doc
}

func (s *pointRangeScorer) Iterator() DocIdSetIterator {
	return s
}

func (s *pointRangeScorer) NextDoc() int {
	return s.Advance(s.doc + 1)
}

func (s *pointRangeScorer) Advance(target int) int {
	for s.pos+1 < len(s.docs) {
		s.pos++
		docID := s.docs[s.pos]
		if docID < target {
			continue
		}
		if s.liveDocs != nil && s.liveDocs.Get(docID) {
			continue
		}
		s.doc = docID
		return docID
	}
	s.doc = NoMoreDocs
	return NoMoreDocs
}

func (s *pointRangeScorer) Cost() int64 {
	return int64(len(s.docs))
}
```

Note: The `Min` and `Max` fields are exported now (capitalized) to support the server layer test from Task 1 of the original plan that accesses `prq.Min` and `prq.Max`. The `field` stays unexported since it's not needed externally.

Add a section noting the known divergences:

```markdown
### Point Fields: BKD Tree Implementation

gosearch implements a 1D BKD tree for point field range queries, matching Lucene's
architecture with the visitor pattern (`IntersectVisitor`, `Relation` enum, `PointTree`
interface). Key divergences from Lucene:

- **1D only** — Lucene supports 1-16 dimensions for geo_point and other multi-dim types.
  Multi-dimensional support tracked in #17.
- **Sort-based partitioning** — Lucene uses BKDRadixSelector for multi-dim efficiency.
  We sort once and split at medians, which is correct and simpler for 1D.
- **Single .kd file** — Lucene uses three files (.poi, .idx, .kdd). We use one file.
- **No prefix compression** — Lucene prefix-codes split values and leaf data.
```
